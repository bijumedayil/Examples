/******* creating stored procedure with name PROC1 *******/

CREATE OR REPLACE PROCEDURE PROC1 (ID NUMBER)
AS 
M1 NUMBER;
M2 NUMBER;
M3 NUMBER;
T NUMBER;
A NUMBER;
BEGIN
SELECT M1,M2,M3 INTO M1,M2,M3 FROM STUDENT WHERE SID=ID;
T := M1+M2+M3;
A := T/3;
UPDATE STUDENT SET TOTAL=T, AVERAGE=A WHERE SID=ID;
END;
