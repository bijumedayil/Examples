//giving alias while using a function

SELECT SUM(SAL) AS TOTAL_SALARY_OF_CLERKS FROM EMP WHERE JOB='CLERK';

SELECT AVG(SAL) AS AVERAGE_SALARY FROM EMP;

// using alias with column name
SELECT ENAME, SAL FROM EMP;


SELECT ENAME AS EMPLOYEE_NAME, SAL AS SALARY FROM EMP;

SELECT ENAME AS EMPLOYEE_NAME, SAL AS SALARY FROM EMP WHERE JOB = 'CLERK';

// using alias with table name

SELECT E.ENAME , E.SAL FROM EMP E;

SELECT E.ENAME AS EMPLOYEE_NAME, E.SAL AS SALARY FROM EMP E;

SELECT E.ENAME, E.SAL FROM EMP E WHERE JOB = 'CLERK';
