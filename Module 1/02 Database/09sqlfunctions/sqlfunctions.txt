// conversion functions

// TO_DATE function (converts string to date)

DROP TABLE EMP_DATE;

CREATE TABLE EMP_DATE(NAME VARCHAR2(30),DOB DATE);

INSERT INTO EMP_DATE(NAME,DOB) VALUES ('AAA',TO_DATE('2011/09/23','YYYY/MM/DD'));

INSERT INTO EMP_DATE(NAME,DOB) VALUES ('BBB',TO_DATE('31/12/2008','DD/MM/YYYY'));

INSERT INTO EMP_DATE(NAME,DOB) VALUES ('CCC',TO_DATE('07-29-03', 'MM-DD-YY'));

INSERT INTO EMP_DATE(NAME,DOB) VALUES ('DDD',TO_DATE('072903', 'MMDDYYYY'));

INSERT INTO EMP_DATE(NAME,DOB) VALUES ('EEE',SYSDATE);

SELECT * FROM EMP_DATE;

SELECT NAME, DOB FROM EMP_DATE WHERE DOB = TO_DATE ('07-29-2003','MM-DD-YYYY');


//TO_CHAR function (converts date to string)

SELECT * FROM EMP_DATE;

SELECT NAME, DOB FROM EMP_DATE;

SELECT NAME, DOB FROM EMP_DATE WHERE NAME='AAA';

SELECT NAME, TO_CHAR(DOB, 'DD-MM-YY') FROM EMP_DATE;

SELECT NAME, TO_CHAR(DOB, 'MM-DD-YYYY') FROM EMP_DATE WHERE NAME='AAA';

SELECT NAME, TO_CHAR(DOB, 'DD/MM/YYYY') FROM EMP_DATE WHERE NAME='AAA';

SELECT NAME, TO_CHAR(DOB, 'YYYY/MM/DD') FROM EMP_DATE WHERE NAME='AAA';


// will display date and time in 12 hour format
SELECT NAME, TO_CHAR(DOB, 'DD-MM-YY HH:MI:SS') FROM EMP_DATE;

// will display date and time in 12 hour format
SELECT NAME, TO_CHAR(DOB, 'DD-MM-YY HH:MI:SS') FROM EMP_DATE WHERE NAME ='EEE';

// will display date and time in 24 hour format
SELECT NAME, TO_CHAR(DOB, 'DD-MM-YY HH24:MI:SS') FROM EMP_DATE;

// will display date and time in 24 hour format
SELECT NAME, TO_CHAR(DOB, 'DD-MM-YY HH24:MI:SS') FROM EMP_DATE WHERE NAME ='EEE';

// NVL function (replaces null values with required values, used for avoiding null values in a column)

DROP TABLE EMP;

CREATE TABLE EMP(EMP_ID NUMBER(5), NAME VARCHAR(30),SALARY NUMBER(5), COMM NUMBER(5));

INSERT INTO EMP(EMP_ID, NAME, SALARY) VALUES (101,'AAA',1000);

INSERT INTO EMP(EMP_ID, NAME, COMM) VALUES (102,'BBB',222);

SELECT NAME, COMM FROM EMP;

// will replace null value with 0
SELECT NAME, NVL(COMM,0) FROM EMP;

// will replace null value with 20
SELECT NAME, NVL(COMM,20) FROM EMP;


// will replace null value with salary
SELECT NAME, NVL(COMM,SALARY) FROM EMP;
